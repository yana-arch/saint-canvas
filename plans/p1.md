### 1ï¸âƒ£ **Image Upload Module**

```jsx
// components/Upload/ImageUploader.jsx
import React, { useCallback } from 'react';
import { useDropzone } from 'react-dropzone';
import { useEditorStore } from '../../store/editorStore';

const ImageUploader = () => {
  const setBaseImage = useEditorStore((state) => state.setBaseImage);
  
  const onDrop = useCallback((acceptedFiles) => {
    const file = acceptedFiles[0];
    const reader = new FileReader();
    
    reader.onload = (e) => {
      setBaseImage({
        src: e.target.result,
        name: file.name,
        size: file.size,
        type: file.type
      });
    };
    
    reader.readAsDataURL(file);
  }, [setBaseImage]);

  const { getRootProps, getInputProps, isDragActive } = useDropzone({
    onDrop,
    accept: {
      'image/*': ['.jpeg', '.jpg', '.png', '.webp']
    },
    maxSize: 10485760, // 10MB
    multiple: false
  });

  return (
    <div 
      {...getRootProps()} 
      className={`upload-zone ${isDragActive ? 'active' : ''}`}
    >
      <input {...getInputProps()} />
      <div className="upload-content">
        <i className="icon-upload" />
        <h3>Upload Image</h3>
        <p>Drag and drop or click to select an image</p>
        <span className="supported-formats">
          Supports: JPG, PNG, WebP (Max 10MB)
        </span>
      </div>
    </div>
  );
};

export default ImageUploader;
```

### 2ï¸âƒ£ **Editor Canvas - The Editing Hub**

```jsx
// components/Editor/EditorCanvas.jsx
import React, { useRef, useEffect, useState } from 'react';
import { Stage, Layer, Image, Transformer } from 'react-konva';
import useImage from 'use-image';
import { useEditorStore } from '../../store/editorStore';

const EditorCanvas = () => {
  const stageRef = useRef(null);
  const { baseImage, layers, selectedLayer, updateLayer, setSelectedLayer } = useEditorStore();
  
  const [dimensions, setDimensions] = useState({
    width: 800,
    height: 600
  });

  // Layer Component
  const LayerItem = ({ layer, isSelected, onSelect }) => {
    const [image] = useImage(layer.src);
    const shapeRef = useRef();
    const transformerRef = useRef();

    useEffect(() => {
      if (isSelected && transformerRef.current && shapeRef.current) {
        transformerRef.current.nodes([shapeRef.current]);
        transformerRef.current.getLayer().batchDraw();
      }
    }, [isSelected]);

    return (
      <>
        <Image
          ref={shapeRef}
          image={image}
          x={layer.x}
          y={layer.y}
          width={layer.width}
          height={layer.height}
          rotation={layer.rotation}
          scaleX={layer.scaleX}
          scaleY={layer.scaleY}
          opacity={layer.opacity}
          draggable
          onClick={() => onSelect(layer.id)}
          onTap={() => onSelect(layer.id)}
          onDragEnd={(e) => {
            updateLayer(layer.id, {
              x: e.target.x(),
              y: e.target.y()
            });
          }}
          onTransformEnd={(e) => {
            const node = shapeRef.current;
            updateLayer(layer.id, {
              x: node.x(),
              y: node.y(),
              rotation: node.rotation(),
              scaleX: node.scaleX(),
              scaleY: node.scaleY()
            });
          }}
        />
        {isSelected && (
          <Transformer
            ref={transformerRef}
            boundBoxFunc={(oldBox, newBox) => {
              if (newBox.width < 20 || newBox.height < 20) {
                return oldBox;
              }
              return newBox;
            }}
          />
        )}
      </>
    );
  };

  return (
    <div className="editor-canvas-wrapper">
      <Stage
        ref={stageRef}
        width={dimensions.width}
        height={dimensions.height}
        className="editor-stage"
      >
        {/* Background Layer */}
        <Layer>
          {baseImage && <BaseImageComponent src={baseImage.src} />}
        </Layer>
        
        {/* Editable Layers */}
        <Layer>
          {layers.map((layer) => (
            <LayerItem
              key={layer.id}
              layer={layer}
              isSelected={selectedLayer === layer.id}
              onSelect={setSelectedLayer}
            />
          ))}
        </Layer>
      </Stage>
      
      {/* Zoom Controls */}
      <ZoomControls stageRef={stageRef} />
    </div>
  );
};

export default EditorCanvas;
```

### 5ï¸âƒ£ **State Management (Zustand)**

```javascript
// store/editorStore.js
import { create } from 'zustand';
import { devtools, persist } from 'zustand/middleware';
import { v4 as uuidv4 } from 'uuid';

export const useEditorStore = create(
  devtools(
    persist(
      (set, get) => ({
        // Base Image
        baseImage: null,
        setBaseImage: (image) => set({ baseImage: image }),
        
        // Layers
        layers: [],
        selectedLayer: null,
        
        addLayer: (asset) => {
          const newLayer = {
            id: uuidv4(),
            type: asset.category,
            assetId: asset.id,
            name: asset.name,
            src: asset.images.default || Object.values(asset.images)[0],
            x: 100,
            y: 100,
            width: 200,
            height: 200,
            rotation: 0,
            scaleX: 1,
            scaleY: 1,
            opacity: 1,
            visible: true,
            locked: false,
            zIndex: get().layers.length
          };
          
          set((state) => ({
            layers: [...state.layers, newLayer],
            selectedLayer: newLayer.id
          }));
        },
        
        updateLayer: (layerId, updates) => {
          set((state) => ({
            layers: state.layers.map((layer) =>
              layer.id === layerId ? { ...layer, ...updates } : layer
            )
          }));
        },
        
        deleteLayer: (layerId) => {
          set((state) => ({
            layers: state.layers.filter((layer) => layer.id !== layerId),
            selectedLayer: state.selectedLayer === layerId ? null : state.selectedLayer
          }));
        },
        
        setSelectedLayer: (layerId) => set({ selectedLayer: layerId }),
        
        reorderLayers: (fromIndex, toIndex) => {
          set((state) => {
            const newLayers = [...state.layers];
            const [movedLayer] = newLayers.splice(fromIndex, 1);
            newLayers.splice(toIndex, 0, movedLayer);
            return { layers: newLayers };
          });
        },
        
        duplicateLayer: (layerId) => {
          const layer = get().layers.find((l) => l.id === layerId);
          if (layer) {
            const duplicated = {
              ...layer,
              id: uuidv4(),
              name: `${layer.name} (copy)`,
              x: layer.x + 20,
              y: layer.y + 20
            };
            set((state) => ({
              layers: [...state.layers, duplicated],
              selectedLayer: duplicated.id
            }));
          }
        },
        
        // History
        history: [],
        historyIndex: -1,
        
        saveToHistory: () => {
          const currentState = {
            layers: get().layers,
            baseImage: get().baseImage
          };
          
          set((state) => ({
            history: [...state.history.slice(0, state.historyIndex + 1), currentState],
            historyIndex: state.historyIndex + 1
          }));
        },
        
        undo: () => {
          const { historyIndex, history } = get();
          if (historyIndex > 0) {
            const prevState = history[historyIndex - 1];
            set({
              layers: prevState.layers,
              historyIndex: historyIndex - 1
            });
          }
        },
        
        redo: () => {
          const { historyIndex, history } = get();
          if (historyIndex < history.length - 1) {
            const nextState = history[historyIndex + 1];
            set({
              layers: nextState.layers,
              historyIndex: historyIndex + 1
            });
          }
        },
        
        // Canvas Settings
        canvasSettings: {
          width: 800,
          height: 600,
          zoom: 1,
          gridVisible: false
        },
        
        updateCanvasSettings: (settings) => {
          set((state) => ({
            canvasSettings: { ...state.canvasSettings, ...settings }
          }));
        },
        
        // Reset
        resetEditor: () => {
          set({
            baseImage: null,
            layers: [],
            selectedLayer: null,
            history: [],
            historyIndex: -1
          });
        }
      }),
      {
        name: 'saint-canvas-storage',
        partialize: (state) => ({
          canvasSettings: state.canvasSettings
        })
      }
    )
  )
);
```
